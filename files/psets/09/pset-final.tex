  \begin{questions}
    \question For each of the following functions $f: \mathbb{Z} \times
      \mathbb{Z} \rightarrow \mathbb{Z}$, determine if the function is
      one-to-one, onto, both, or none.

      \begin{parts}
        \part $f(x,y)=2x-4y$
          \begin{solution}
          \end{solution}
          \vspace{\stretch{1}}
        \part $f(x,y)=(x+1,2y)$
          \begin{solution}
            One-to-one, but not onto. There is no pair $(x,y)$ such that
            $f(x,y)=(1,1)$.
          \end{solution}
          \vspace{\stretch{1}}
        \part $f(x,y)=(1-y,1-x)$
          \begin{solution}
            One-to-one and onto.
          \end{solution}
          \vspace{\stretch{1}}
      \end{parts}

    \question For a function $f: A \rightarrow X$, what can we say about the
      relationship between the cardinality of the domain and the target if $f$
      has the following properties.

      \begin{parts}
        \part Onto
          \begin{solution}
            $|A| \ge |X|$
          \end{solution}
          \vspace{\stretch{1}}
        \part One-to-one
          \vspace{\stretch{1}}
          \begin{solution}
            $|A| \le |X|$
          \end{solution}
        \part One-to-one correspondence
          \vspace{\stretch{1}}
          \begin{solution}
            $|A| = |X|$
          \end{solution}
      \end{parts}

    \question For each of the properties, determine a function $f: \mathbb{Z}
      \rightarrow \mathbb{Z}$, that satisfies the properties.

      \begin{parts}
        \part Neither one-to-one nor onto.
          \begin{solution}
            $f(x)=x^2$
          \end{solution}
          \vspace{\stretch{1}}
        \part Onto, but not one-to-one.
          \begin{solution}
            $f(x)=|x|$.
          \end{solution}
          \vspace{\stretch{1}}
        \bonuson
        \part One-to-one, but not onto.
          \begin{solution}
          \end{solution}
          \vspace{\stretch{1}}
        \bonusoff
      \end{parts}
  \end{questions}
